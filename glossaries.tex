% Example glossary entries (add more as needed)
\newglossaryentry{anomaly}{name={anomaly},description={observation or pattern that deviates significantly from a defined notion of normality}}

\newglossaryentry{precision}{%
	name={precision},%
	description={for anomaly detection, the proportion of predicted anomalous points or segments that are actually anomalous (true positives divided by all positive predictions)}%
}

\newglossaryentry{recall}{%
	name={recall},%
	description={for anomaly detection, the proportion of truly anomalous points or segments that are correctly detected (true positives divided by all actual anomalies)}%
}

\newglossaryentry{ground-truth}{%
	name={ground truth},%
	description={reference labels that indicate for each observation whether it is considered normal or anomalous, used as a standard when evaluating detection performance}%
}

\newglossaryentry{confusion-matrix}{%
	name={confusion matrix},%
	description={tabular summary of prediction results that counts true positives, true negatives, false positives, and false negatives}%
}

\newglossaryentry{benchmark}{%
	name={benchmark},%
	description={standardized dataset and evaluation protocol used to compare the performance of different anomaly detection methods}%
}

\newglossaryentry{mislabeling}{%
	name={mislabeling},%
	description={inconsistent assignment of anomaly labels to similar or identical patterns, which distorts evaluation by inflating false-negative rates}%
}

\newglossaryentry{run-to-failure-bias}{%
	name={run-to-failure bias},%
	description={systematic placement of anomalies at the end of a time series, which favors models that exploit positional cues rather than genuine signal deviations}%
}

\newglossaryentry{unrealistic-anomaly-ratio}{%
	name={unrealistic anomaly ratio},%
	description={an artificially high proportion of anomalous observations in a dataset compared to real-world systems, which can lead to over-optimistic performance estimates}%
}
